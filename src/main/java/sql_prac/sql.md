// 동명 동물 수 찾기   
SELECT NAME,
COUNT(NAME) AS COUNT
FROM ANIMAL_INS
GROUP BY 1
HAVING COUNT > 1
ORDER BY NAME

//아픈 동물 찾기
SELECT ANIMAL_ID, NAME
FROM ANIMAL_INS
WHERE INTAKE_CONDITION = "Sick"

//상위 n개 레코드
SELECT NAME
FROM ANIMAL_INS
ORDER BY DATETIME
LIMIT 5

//최솟값 찾기
SELECT MIN(DATETIME) AS "시간"
FROM ANIMAL_INS

//젊은 동물 찾기
SELECT ANIMAL_ID, NAME
FROM ANIMAL_INS
WHERE INTAKE_CONDITION <> "Aged"
ORDER BY ANIMAL_ID
